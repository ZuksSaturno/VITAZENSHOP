function updateCartUI() {

async function finalizarPedido() {
  const nome     = document.getElementById('fNome').value.trim();
  const email    = document.getElementById('fEmail').value.trim();
  const telefone = document.getElementById('fTelefone').value.trim();
  const cpf      = document.getElementById('fCpf').value.trim();
  const rua      = document.getElementById('fRua').value.trim();
  const bairro   = document.getElementById('fBairro').value.trim();
  const cidade   = document.getElementById('fCidade').value.trim();
  const estado   = document.getElementById('fEstado').value.trim();
  const cep      = document.getElementById('fCep').value.trim();

  // Validação básica
  const required = [{id:'fNome',v:nome},{id:'fEmail',v:email},{id:'fTelefone',v:telefone},{id:'fRua',v:rua},{id:'fCidade',v:cidade},{id:'fCep',v:cep}];
  let valid = true;
  required.forEach(({id,v})=>{
    const el=document.getElementById(id);
    if(!v){ el.classList.add('error'); valid=false; } else { el.classList.remove('error'); }
  });
  if(!valid){ showToast('⚠️ Preencha todos os campos obrigatórios'); return; }

  const entries = Object.entries(cart).filter(([,q])=>q>0);
  if(!entries.length){ showToast('Carrinho vazio!'); return; }

  // UI de loading
  const btn = document.getElementById('checkoutBtn');
  const label = document.getElementById('checkoutLabel');
  const spinner = document.getElementById('spinner');
  btn.disabled = true;
  label.textContent = 'Processando...';
  spinner.style.display = 'block';

  const itens = entries.map(([id,qty])=>{
    const p=products.find(x=>x.id==id);
    return { id:p.id, nome:p.name, descricao:p.desc, qty, preco:p.price };
  });

  const total = itens.reduce((s,i)=>s+(i.preco*i.qty),0);

  try {
    const res = await fetch('/api/criar-pagamento', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        itens, total,
        pagamento: selectedPayment,
        cliente: { nome, email, telefone, cpf, endereco:rua, bairro, cidade, estado, cep }
      })
    });

    const data = await res.json();

    if(!res.ok || !data.url){
      throw new Error(data.error || 'Erro ao gerar cobrança');
    }

    // Redireciona para página de pagamento do AbacatePay
    window.location.href = data.url;

  } catch(err) {
    showToast('❌ ' + (err.message || 'Erro ao processar. Tente novamente.'));
    btn.disabled = false;
    label.textContent = '🔒 Ir para o pagamento';
    spinner.style.display = 'none';
  }
}
